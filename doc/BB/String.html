def strip_ansi(text)
  text.gsub(/\x1b(\[|\(|\))[;?0-9]*[0-9A-Za-z]/, '')
      .gsub(/\x1b(\[|\(|\))[;?0-9]*[0-9A-Za-z]/, '')
      .gsub(/(\x03|\x1a)/, '')
end
